# Routes
# This file defines all application routes (Higher priority routes first)
# ~~~~

# Home page
GET     /                           controllers.Application.index
POST    /characters/create               controllers.Application.createOrUpdateCharacter
GET     /characters/:id                  controllers.Application.findUserById(id: Long)
PUT     /characters/:id                  controllers.Application.updateUser(id: Long)
DELETE  /characters/:id                  controllers.Application.deleteUser(id: Long)

# Map static resources from the /public folder to the /assets URL path
GET     /assets/*file               controllers.Assets.at(path="/public", file)
